class Solution:
    def removeDuplicates(self, nums: List[int]) -> int:
        n=len(nums)
        i=0
        j=1
        if(n==1):
           return 1
        while(j<n):
            if(nums[i]!=nums[j]):
                i=i+1
                nums[i]=nums[j]
                
            j=j+1
        return i+1
