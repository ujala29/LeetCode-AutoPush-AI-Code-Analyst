class Solution:
    def reverseWords(self, s: str) -> str:
        ans=""
        i=0
        while i<len(s):
            j=i
            while j <len(s) and s[j]!=" ":
                j=j+1
            

            ans+=s[i:j][::-1]
            if j<len(s):
                 ans+=" "
            i=j+1
        
        return ans
