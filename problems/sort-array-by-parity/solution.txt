class Solution:
    def sortArrayByParity(self, nums: List[int]) -> List[int]:
        j=len(nums)-1
        i=0
        ans=[0]*len(nums)
        for m in range(0,len(nums)):
            if nums[m]%2==0:
                ans[i]=nums[m]
                i=i+1
            else:
                ans[j]=nums[m]
                j=j-1
        

        return ans
